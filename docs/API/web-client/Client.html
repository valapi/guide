Config

interface Config {
    /**
     * Region
     */
    region?: string;

    /**
     * Client Config
     */
    client?: {
        version?: string;
        platform?: {
            "platformType": string;
            "platformOS": string;
            "platformOSVersion": string;
            "platformChipset": string;
        };
    };

    /**
     * Request Config
     * Config for RequestClient (Axios)
     */
    axiosConfig?: AxiosRequestConfig;

    /**
     * Expire time
     * The time in milliseconds that data will be refreshed.
     */
    expiresIn?: {
        cookie: number;
        token?: number;
    };

    /**
     * Throw the Error?
     */
    throwOnError?: boolean;
}
